#include <iostream>
#include <algorithm>
using namespace std;

int numbers[11];
int numMap[6][5] = { {-1, -1, -1, -1, -1},
                     {-1,  1,  2,  3, -1},
                     {-1,  4,  5,  6, -1},
                     {-1,  7,  8,  9, -1},
                     {-1, -1,  0, -1, -1},
                     {-1, -1, -1, -1, -1} };

typedef struct
{
    int x, y;
}Point;

Point findP(int x)
{
    Point p;
    int status = 0;
    for (int i = 0; i < 6; ++i)
    {
        for (int j = 0; j < 5; ++j)
        {
            if (numMap[i][j] == x)
            {
                p.x = i;
                p.y = j;
                status = 1;
                break;
            }
        }
        if(status == 1)
            break;
    }
    return p;
}

bool judge(int num[], int k)
{
    Point p[k+1];
    for (int i = 0; i < k; ++i)
    {
        p[i] = findP(num[i]);
    }
    int okay[4] = { 0 };
    //左移
    for (int i = 0; i < k; ++i)
    {
        if (numMap[p[i].x-1][p[i].y] == -1)
            okay[0] = 1;
    }
    //右移
    for (int i = 0; i < k; ++i)
    {
        if (numMap[p[i].x+1][p[i].y] == -1)
            okay[1] = 1;
    }
    //上移
    for (int i = 0; i < k; ++i)
    {
        if (numMap[p[i].x][p[i].y-1] == -1)
            okay[2] = 1;
    }
    //下移
    for (int i = 0; i < k; ++i)
    {
        if (numMap[p[i].x][p[i].y+1] == -1)
            okay[3] = 1;
    }
    int ans = 0;
    for (int i = 0; i < 4; ++i)
    {
        ans += okay[i];
    }
    if (ans == 4)
        return true;
    return false;
}

int main(int argc, const char * argv[])
{
    int K;
    string s;
    while(cin >> K)
    {
        cin >> s;
        for (int i = 0; i < s.length(); ++i)
        {
            numbers[i] = s[i]-'0';
        }
        if (judge(numbers, K))
        {
            cout << "YES" << endl;
        }
        else
        {
            cout << "NO" << endl;
        }
        
    }
    return 0;
}
